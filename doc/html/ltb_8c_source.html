343 //get min max values for interpolation data
344 int get_min_max_for_interp(const Ltb_model *model, double *max, double *min)
345 {
346  unsigned int i,j;
347  double max_t = model->t_vec->data[0] - model->tB_vec->data[0];
348  double min_t = model->t_vec->data[0] - model->tB_vec->data[0];
349  double tmp=0;
350 
351  //find max value of t - tB
352  for (i=0;i<model->t_vec->size;++i)
353  {
354  for (j=0;j<model->tB_vec->size;++j)
355  {
356  tmp = model->t_vec->data[i] - model->tB_vec->data[j];
357  if (tmp > max_t)
358  max_t = tmp;
359  if (tmp < min_t)
360  min_t = tmp;
361  }
362  }
363  //assume that M_max is always ge 0
364  double M_max = gsl_vector_max(model->M_vec);
365  double M_min = gsl_vector_min(model->M_vec);
366  double chi_max = 1;
367  double chi_min = 1;
368  if (model->cur == 0)
369  {
370  chi_max = 1;
371  chi_min = 1;
372  }
373  else
374  {
375  if (model->cur > 0)
376  {
377  //E_vec is < 0
378  chi_max = -2.0 * gsl_vector_min(model->E_vec);
379  chi_min = -2.0 * gsl_vector_max(model->E_vec);
380  }
381  else
382  {
383  //E_vec is > 0
384  chi_max = 2.0 * gsl_vector_max(model->E_vec);
385  chi_min = 2.0 * gsl_vector_min(model->E_vec);
386  }
387  chi_max = pow(chi_max,1.5);
388  chi_min = pow(chi_min,1.5);
389  }
390  *max = max_t*chi_max / (G_grav*M_min);
391  *min = min_t*chi_min / (G_grav*M_max);
392 
393  return GSL_SUCCESS;
394 }

828 static int check_slope_R_t(double r, double t, double t_val, const Ltb_model *model, Interp_data *eta_interp)
829 {

853  }
